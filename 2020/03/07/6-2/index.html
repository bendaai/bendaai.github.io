用栈模拟出栈顺序

题目链接

https://vjudge.net/problem/UVA-514

import sys
def _input():
    lines=sys.stdin.read().split('\n')
    for val in lines:
        x=val.strip()
        if x!='':
            yield val
c=_input()
while 1:
    n=int(next(c))
    if n==0:
        break
    while 1:
        sta=[]
        pop_list=list(map(int,next(c).split()))
        if pop_list[0]==0:
            print()
            break
        k=0
        i=1
        while i<=n:
            if len(sta)==0 or pop_list[k]!=sta[-1]:
                sta.append(i)
                i+=1
            else:
                sta.pop()
                k+=1
        while len(sta)>0 and sta[-1]==pop_list[k]:
            k+=1
            sta.pop()
        if len(sta)>0:
            print('No')
        else:
            print('Yes')
